set -- "$@" --tmp-- .cmi --tmp-- .cmo --tmp-- .byte --suffix .ml
SH=${SH:-sh -eu}; export SH
exec nofake-exec.sh --error -Rprog "$@" -- ${SH} -c '
    ocamlc -o "${1%.ml}.byte" "${1}"
    ocamlrun "${1%.ml}.byte"
    exec "${1%.ml}.byte"
' --
exit 1
