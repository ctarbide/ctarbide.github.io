for i in "${1}" hello-*.sh; do
    i=${i#./}
    if [ ! -e "${i}.html" -o "${i}" -nt "${i}.html" ]; then
        ( <<htmlify "${i}">> ) | <<md-autoheader-autolink.pl>> |
            <<assets - md.pl for pages>> >"${i}.html"
    fi
done
